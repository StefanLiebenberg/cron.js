For this example you will need a browser that supports console.log. Please take a look there at the output 

var Crontab = new Cron();
Crontab.start();

Crontab.add(new Cron.Job( "@second" , function () {
  console.log( new Date().getSeconds() )
  console.log( "This message should run every second" )
}));

Crontab.add( new Cron.Job( "*/5 * * * * *", function () {
  console.log( "This message should run every 5 seconds" );
}));

Crontab.add( new Cron.Job( "0-5,10-20,40-59/2 * * * * *", function () {
  console.log( "This message will on seconds 0 to 5, then on 10 to 20 and then 2nd second after 40 " )
}));
